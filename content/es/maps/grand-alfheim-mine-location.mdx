  <Step num="1" title="Acepta la misión del herrero">

  <Step num="3" title="Elimina a las criaturas guardianas hostiles">

  </Step>
  <Step num="4" title="Cosecha y extrae minerales">

  <Step num="5" title="Forja tu equipamiento mejorado">
